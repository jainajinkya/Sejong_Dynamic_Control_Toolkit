FILE(GLOB_RECURSE header *.h)
file(GLOB_RECURSE source *.cpp)
add_executable (simulation_NAO
${header} ${source}
  )

target_link_libraries (simulation_NAO
  common
  LieGroup
  srDyn
  srg
  Renderer
  util
  SR_SysGenerator
  NAO_Controller
  SR_Terrain
)

endif (APPLE)
